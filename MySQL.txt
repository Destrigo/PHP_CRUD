CREATE DATABASE problem_solver;

USE problem_solver;

CREATE TABLE problems (
  id INT AUTO_INCREMENT PRIMARY KEY,
  ADD user_id INT NOT NULL,
  title VARCHAR(255) NOT NULL,
  description TEXT,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE solutions (
  id INT AUTO_INCREMENT PRIMARY KEY,
  ADD user_id INT NOT NULL,
  problem_id INT NOT NULL,
  content TEXT NOT NULL,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (problem_id) REFERENCES problems(id) ON DELETE CASCADE
);

CREATE TABLE users (
    id INT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(50) NOT NULL UNIQUE,
    email VARCHAR(100) NOT NULL UNIQUE,
    password VARCHAR(255) NOT NULL,  -- hashed password
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE votes (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    entity_type ENUM('problem','solution') NOT NULL,
    entity_id INT NOT NULL,
    UNIQUE KEY unique_vote (user_id, entity_type, entity_id),
    FOREIGN KEY (user_id) REFERENCES users(id)
);

ALTER TABLE problems ADD rating INT DEFAULT 0;
ALTER TABLE solutions ADD rating INT DEFAULT 0;
